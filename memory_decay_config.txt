# OrKa Memory Decay Configuration Reference
# ========================================

# ⚠️ IMPORTANT: Environment Variables Override YAML Configuration
# If you see TTL values like 0.1h/0.2h in logs instead of expected 2h/168h,
# check for conflicting environment variables that override your YAML settings.

# To check current settings:
# orka memory configure

# OPTION 1: Disable memory decay completely (RECOMMENDED for development)
set ORKA_MEMORY_DECAY_ENABLED=false

# OPTION 2: Enable with production settings (matches README)
set ORKA_MEMORY_DECAY_ENABLED=true
set ORKA_MEMORY_DECAY_SHORT_TERM_HOURS=2       # Short-term: 2 hours (matches README)
set ORKA_MEMORY_DECAY_LONG_TERM_HOURS=168      # Long-term: 1 week (matches README)  
set ORKA_MEMORY_DECAY_CHECK_INTERVAL_MINUTES=30  # Check every 30 minutes

# OPTION 3: Development settings (longer retention for testing)
set ORKA_MEMORY_DECAY_ENABLED=true
set ORKA_MEMORY_DECAY_SHORT_TERM_HOURS=24     # Keep logs for 24 hours
set ORKA_MEMORY_DECAY_LONG_TERM_HOURS=168     # Keep important memories for 1 week  
set ORKA_MEMORY_DECAY_CHECK_INTERVAL_MINUTES=60  # Check every hour

# OPTION 4: Fast decay for testing memory expiration
set ORKA_MEMORY_DECAY_ENABLED=true
set ORKA_MEMORY_DECAY_SHORT_TERM_HOURS=0.1    # 6 minutes (for testing only)
set ORKA_MEMORY_DECAY_LONG_TERM_HOURS=0.2     # 12 minutes (for testing only)
set ORKA_MEMORY_DECAY_CHECK_INTERVAL_MINUTES=5   # Check every 5 minutes

# OPTION 5: Production settings (balanced retention)
set ORKA_MEMORY_DECAY_ENABLED=true
set ORKA_MEMORY_DECAY_SHORT_TERM_HOURS=8      # 8 hours for logs
set ORKA_MEMORY_DECAY_LONG_TERM_HOURS=72      # 3 days for important data
set ORKA_MEMORY_DECAY_CHECK_INTERVAL_MINUTES=120  # Check every 2 hours

# To use RedisStack on port 6380 (if needed):
set REDIS_URL=redis://localhost:6380/0
